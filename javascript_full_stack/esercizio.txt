la potenza di filter, map e reduce fa una pipeline


const prodotti = [
    { nome: "A", prezzo: 10, attivo: true },
    { nome: "B", prezzo: 20, attivo: false },
    { nome: "C", prezzo: 30, attivo: true }
];

const totale = prodotti
    .filter(p => p.attivo)
    .map(p => ({ ...p, prezzo: p.prezzo * 1.1 }))
    .reduce((acc, p) => acc + p.prezzo, 0);

console.log(totale);
